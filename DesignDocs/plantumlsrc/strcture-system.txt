  node "评测机" as judger
  control "评测控制端" as judgerbackend
  folder "评测控制端文件系统" as jbfs {
    folder 评测文件[

      ===

    ]
  }
}

judgerbackend --> webbackend : 返回评测结果
judger --> judgerbackend : 返回评测结果
judger --> judgerbackend : 返回评测机状态
